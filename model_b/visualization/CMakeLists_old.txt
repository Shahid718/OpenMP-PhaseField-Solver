#-------------------------------------------------------------------------------
#  CMake Configuration for Cahn-Hilliard Phase-Field Simulation
#-------------------------------------------------------------------------------

cmake_minimum_required(VERSION 3.15)

#-------------------------------------------------------------------------------
#  Project Definition
#-------------------------------------------------------------------------------

project(CahnHilliard
    VERSION 1.0.0
    LANGUAGES Fortran
    DESCRIPTION "Cahn-Hilliard Phase-Field Simulation Suite"
)

#-------------------------------------------------------------------------------
#  Build Options
#-------------------------------------------------------------------------------

option(WITH_OPENMP "Enable OpenMP parallelization" ON)
option(USE_OPENMP_MACRO "Define USE_OPENMP preprocessor macro" ON)

set(CMAKE_BUILD_TYPE "Release" CACHE STRING "Build type: Debug, Release, RelWithDebInfo, MinSizeRel")

#-------------------------------------------------------------------------------
#  Set Compiler (gfortran)
#-------------------------------------------------------------------------------

set(CMAKE_Fortran_COMPILER gfortran CACHE STRING "Fortran compiler" FORCE)

if(NOT CMAKE_Fortran_COMPILER_ID MATCHES "GNU")
    message(WARNING "This project is optimized for gfortran compiler.")
    message(WARNING "Current compiler: ${CMAKE_Fortran_COMPILER_ID}")
endif()

#-------------------------------------------------------------------------------
#  Compiler Flags (gfortran)
#-------------------------------------------------------------------------------

# Base flags
set(CMAKE_Fortran_FLAGS "-cpp -ffree-line-length-none -Wall -Wextra -Wimplicit-interface -Wunused-parameter -Wmaybe-uninitialized")

# Debug flags
if(CMAKE_BUILD_TYPE STREQUAL "Debug")
    set(CMAKE_Fortran_FLAGS "${CMAKE_Fortran_FLAGS} -O0 -g -fbacktrace -fbounds-check -fcheck=all -ffpe-trap=invalid,zero,overflow")
    message(STATUS "Build Type: DEBUG")
    
# Release with Debug Info
elseif(CMAKE_BUILD_TYPE STREQUAL "RelWithDebInfo")
    set(CMAKE_Fortran_FLAGS "${CMAKE_Fortran_FLAGS} -O2 -g -fbacktrace")
    message(STATUS "Build Type: RELEASE WITH DEBUG INFO")
    
# Minimum Size Release
elseif(CMAKE_BUILD_TYPE STREQUAL "MinSizeRel")
    set(CMAKE_Fortran_FLAGS "${CMAKE_Fortran_FLAGS} -Os")
    message(STATUS "Build Type: MINIMUM SIZE RELEASE")
    
# Release (default)
else()
    # REMOVED invalid flag: -fno-fpe-trap=invalid,zero,overflow
    set(CMAKE_Fortran_FLAGS "${CMAKE_Fortran_FLAGS} -O3 -march=native -funroll-loops -ftree-vectorize -fstack-arrays")
    message(STATUS "Build Type: RELEASE")
endif()

#-------------------------------------------------------------------------------
#  OpenMP Configuration
#-------------------------------------------------------------------------------

if(WITH_OPENMP)
    find_package(OpenMP)
    if(OpenMP_Fortran_FOUND)
        set(CMAKE_Fortran_FLAGS "${CMAKE_Fortran_FLAGS} ${OpenMP_Fortran_FLAGS}")
        set(USE_OPENMP_MACRO ON)
        message(STATUS "OpenMP: ENABLED")
        message(STATUS "OpenMP Flags: ${OpenMP_Fortran_FLAGS}")
    else()
        message(WARNING "OpenMP not found. Building without OpenMP.")
        set(WITH_OPENMP OFF)
        set(USE_OPENMP_MACRO OFF)
    endif()
else()
    message(STATUS "OpenMP: DISABLED")
endif()

#-------------------------------------------------------------------------------
#  Preprocessor Definitions
#-------------------------------------------------------------------------------

if(USE_OPENMP_MACRO AND WITH_OPENMP)
    set(CMAKE_Fortran_FLAGS "${CMAKE_Fortran_FLAGS} -DUSE_OPENMP")
    message(STATUS "Preprocessor: USE_OPENMP defined")
endif()

#-------------------------------------------------------------------------------
#  Operating System Specific Settings
#-------------------------------------------------------------------------------

if(WIN32)
    message(STATUS "Operating System: Windows")
    set(CMAKE_Fortran_FLAGS "${CMAKE_Fortran_FLAGS} -fno-underscoring")
    if(NOT CMAKE_GENERATOR)
        set(CMAKE_GENERATOR "MinGW Makefiles" CACHE STRING "Generator" FORCE)
    endif()
elseif(UNIX)
    message(STATUS "Operating System: Linux/Unix")
    set(CMAKE_Fortran_FLAGS "${CMAKE_Fortran_FLAGS} -fPIC")
else()
    message(STATUS "Operating System: Other")
endif()

#-------------------------------------------------------------------------------
#  Source Directories
#-------------------------------------------------------------------------------

set(SRC_DIR "${CMAKE_SOURCE_DIR}/src")
set(APP_DIR "${CMAKE_SOURCE_DIR}/app")

#-------------------------------------------------------------------------------
#  Source Files
#-------------------------------------------------------------------------------

set(MODULE_SOURCES
    ${SRC_DIR}/precisions_control.f90
    ${SRC_DIR}/error_handler.f90
    ${SRC_DIR}/performance_timer.f90
    ${SRC_DIR}/core_utilities.f90
    ${SRC_DIR}/phase_field_solver.f90
    ${SRC_DIR}/phase_field_setup.f90
    ${SRC_DIR}/grid_generator.f90
    ${SRC_DIR}/microstructure_initializer.f90
    ${SRC_DIR}/free_energy_kernel.f90
    ${SRC_DIR}/laplacian_kernel.f90
    ${SRC_DIR}/time_kernel.f90
    ${SRC_DIR}/io_handlers.f90
    ${SRC_DIR}/performance_analyzer.f90
    ${SRC_DIR}/performance_evaluation.f90
)

set(MAIN_SOURCES
    ${APP_DIR}/main_driver.f90
)

#-------------------------------------------------------------------------------
#  Module Directory
#-------------------------------------------------------------------------------

set(MODULE_DIR "${CMAKE_BINARY_DIR}/modules")
file(MAKE_DIRECTORY ${MODULE_DIR})

include_directories(${CMAKE_BINARY_DIR})
include_directories(${MODULE_DIR})

#-------------------------------------------------------------------------------
#  Create Executable
#-------------------------------------------------------------------------------

set(EXECUTABLE_NAME "cahn_hilliard_driver")

add_executable(${EXECUTABLE_NAME} ${MAIN_SOURCES} ${MODULE_SOURCES})

set_target_properties(${EXECUTABLE_NAME} PROPERTIES
    Fortran_MODULE_DIRECTORY ${MODULE_DIR}
    RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin
    ARCHIVE_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/lib
    LIBRARY_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/lib
)

#-------------------------------------------------------------------------------
#  Linker Flags
#-------------------------------------------------------------------------------

if(WITH_OPENMP AND OpenMP_Fortran_FOUND)
    target_link_libraries(${EXECUTABLE_NAME} ${OpenMP_Fortran_FLAGS})
endif()

#-------------------------------------------------------------------------------
#  Windows-specific Settings
#-------------------------------------------------------------------------------

if(WIN32)
    target_link_libraries(${EXECUTABLE_NAME} "-Wl,--stack,10000000")
endif()

#-------------------------------------------------------------------------------
#  Installation
#-------------------------------------------------------------------------------

install(TARGETS ${EXECUTABLE_NAME}
    RUNTIME DESTINATION bin
    LIBRARY DESTINATION lib
    ARCHIVE DESTINATION lib
)

#-------------------------------------------------------------------------------
#  Print Configuration Summary
#-------------------------------------------------------------------------------

message(STATUS "")
message(STATUS "================================================================================")
message(STATUS "  CAHN-HILLIARD PHASE-FIELD SIMULATION - CONFIGURATION SUMMARY")
message(STATUS "================================================================================")
message(STATUS "  Project Name       : ${PROJECT_NAME}")
message(STATUS "  Version            : ${PROJECT_VERSION}")
message(STATUS "  Build Type         : ${CMAKE_BUILD_TYPE}")
message(STATUS "  Compiler           : ${CMAKE_Fortran_COMPILER_ID} ${CMAKE_Fortran_COMPILER_VERSION}")
message(STATUS "  Compiler Flags     : ${CMAKE_Fortran_FLAGS}")
message(STATUS "  OpenMP             : ${WITH_OPENMP}")
message(STATUS "  USE_OPENMP Macro   : ${USE_OPENMP_MACRO}")
message(STATUS "  Executable         : ${EXECUTABLE_NAME}")
message(STATUS "  Module Directory   : ${MODULE_DIR}")
message(STATUS "  Source Directory   : ${CMAKE_SOURCE_DIR}")
message(STATUS "  Build Directory    : ${CMAKE_BINARY_DIR}")
message(STATUS "  Operating System   : ${CMAKE_SYSTEM_NAME}")
if(WIN32)
    message(STATUS "  Generator          : ${CMAKE_GENERATOR}")
endif()
message(STATUS "================================================================================")
message(STATUS "")
